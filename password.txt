import java.util.Scanner;

public class Main {
	public static void main(String []args) {
		int passlen = 8, upper = 0, lower = 0, special = 0, digits = 0;
		char ch;
		Scanner s = new Scanner(System.in);
		System.out.println("\n--------------Rules to Create Password---------------");
		System.out.println("1. The password must be at least 8 characters long");
		System.out.println("2. The password must contain at least 1 uppercase character");
		System.out.println("3. The password must contain at least 1 lowercase character");
		System.out.println("4. The password must contain at least 1 digit between 0-9");
		System.out.println("5. The password must contain at least 1 special character");
		System.out.println("6. The password must not contain > and <");
		System.out.print("\nCreate a password: ");
		String pass = s.nextLine();
		int total = pass.length();

		if (total < passlen) {
			System.out.println("The password should be 8 characters or more");
			return;
		} else {
			for (int i=0; i<total; i++) {
				ch = pass.charAt(i);
				if (Character.isUpperCase(ch)) {
					upper++;
				} else if (Character.isLowerCase(ch)) {
					lower++;
				} else if (Character.isDigit(ch)) {
					digits++;
				} else {
					if (ch == '<' || ch == '>') {
						System.out.println("The password is malicious");
						return;
					} else {
						special++;
					}
				}
			}
		}

		if (upper != 0 && lower != 0 && digits !=0 && special != 0) {
			if (total >= 12) {
				System.out.println("\nThe strength of password is stong");
			} else {
				System.out.println("\nThe strength of password is medium");
			}
			System.out.println("\nThe password contains: ");
			System.out.println(upper + " upper case characters, " + lower + " lower case characters, ");
			System.out.println(digits + " digits, " + special + " special characters, ");
		} else {
			if (upper == 0) {
				System.out.println("The password must contain upper case characters");
			} 
			if (lower == 0) {
				System.out.println("The password must contain lower case characters");
			} 
			if (digits == 0) {
				System.out.println("The password must contain digits");
			} 
			if (special == 0) {
				System.out.println("The password must contain special characters");
			}
		}
	}
}
